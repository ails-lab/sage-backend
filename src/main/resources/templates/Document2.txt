{\large Εισαγωγή}
\begin{itemize}
\item Η κλασική λογική ασχολείται με την αλήθεια των λογικών προτάσεων.
\begin{itemize}
\item Μία πρόταση μπορεί να είναι είτε {\color{blue}αληθής} είτε {\color{blue}ψευδής}.\\
{\color{red} O Νίκος είναι φοιτητής. Όλοι οι φοιτητές έχουν τελειώσει το σχολείο.}
\item Στόχος των διαδικασιών συλλογιστικής είναι η απόδειξη της αλήθειας ή μη κάποιας πρότασης.
\end{itemize}
\item Στον πραγματικό κόσμο πολλές φορές τα πράγματα είναι πιο σχετικά και δεν μπορούμε να πούμε ότι κάτι έχει {\color{blue}πάντα}, {\color{blue}αναμφίβολα}, {\color{blue}απόλυτα}, {\color{blue}πλήρως} και {\color{blue}ακριβώς} μια ιδιότητα.
\begin{itemize}
\item Μια ιδιότητα μπορεί να ισχύει {\it συνήθως} ή {\it σπάνια}. Ύπαρξη εξαιρέσεων.\\
{\color{red} Τα περισσότερα πτηνά πετάνε, αλλά όχι όλα.}
\item Μπορεί να μην είμαστε απόλυτα βέβαιοι για κάτι.\\
{\color{red} Ο Νίκος μάλλον έχει γρίπη.}
\begin{itemize}
\item Η τιμή μιας ιδιότητας μπορεί να υπολογιστεί με κάποια {\it ανακρίβεια}.\\[-26pt]
%{\bf Η θερμοκρασία μπορεί να μετρηθεί με κάποιο επίπεδο ακρίβειας.}
\item Μπορεί να γνωρίζουμε κάτι από μια {\it μερικώς αξιόπιστη} πηγή.
\end{itemize}
\item Μια ιδιότητα μπορεί να ισχύει {\it μερικώς}.\\
{\color{red} Η Κατερίνα είναι ψηλή, άλλα όχι πολύ.}
\end{itemize}
\end{itemize}

\begin{slide}
\begin{small}
{\large Η έννοια της πιθανότητας}
\begin{itemize}
\item Εν γένει, η πιθανότητα είναι ένα {\it μέτρο} του πόσο πιθανό είναι να συμβεί/ισχύει κάποιο γεγονός.
\item Όσο μεγαλύτερη είναι η πιθανότητα τόσο πιο πιθανό είναι να συμβεί/ισχύει το γεγονός.
\item {\it Φυσική} ή {\it αντικειμενική} ερμηνεία:
\begin{itemize}
\item Σε πειράματα τύχης ή σε φυσικές διεργασίες που εμπεριέχουν τυχαιότητα τα επιμέρους γεγονότα τείνουν να εμφανίζονται με {\it σταθερό ρυθμό} σε {\it βάθος χρόνου}.
\item H πιθανότητα μπορεί να ιδωθεί ως η {\it σχετική συχνότητα} εμφάνισης του γεγονότος ή ως η {\it φυσική τάση} μιας κατάστασης να οδηγεί σε κάποιο γεγονός.
\end{itemize}
\item {\it Εμπειρική} ή {\it υποκειμενική} ερμηνεία:
\begin{itemize}
\item Η πιθανότητα οποιασδήποτε πρότασης μπορεί να θεωρηθεί ως η υποκειμενική αληθοφάνειά της ή ο βαθμός με τον οποίο τα στοιχεία που διαθέτουμε στηρίζουν την ισχύ της πρότασης.
\item {\it Μπεϋζιανή πιθανότητα}: Η πιθανότητα μιας πρότασης είναι ο βαθμός πίστης μας σε αυτή την πρόταση.
\end{itemize}
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Αντικειμενική πιθανότητα}
\begin{itemize}
\item Οι αντικειμενικές πιθανότητες αντιπροσωπεύουν {\it συχνότητα}.
\item Η πιθανότητα $\pr(a)$ που αποδίδουμε στο να συμβεί ένα γεγονός $a$ είναι\\[3pt]
 \hbox to 1cm{}\vbox{το {\it ποσοστό των φορών} που αυτό συμβαίνει\\
  σε {\it μια ακολουθία πολλών γεγονότων}\\
   όταν επαναλαμβάνεται η {\it ίδια διαδικασία} υπό τις ίδιες συνθήκες\\
    και κάθε γεγονός είναι {\it ανεξάρτητο} από τα προηγούμενα.}
\item Κλασικά παραδείγματα:
\begin{itemize}
\item Ρίψη κέρματος, επιλογή τραπουλόχαρτου.
\end{itemize}
\item Η πιθανότητα καλείται {\it αντικειμενική} γιατί δεν εξαρτάται από κάποια προσωπική μας εκτίμηση.
\item Πρακτικός υπολογισμός: $\displaystyle \pr(a) \doteq \lim_{n\rightarrow+\infty}\frac{n_a}{n}$.
\begin{itemize}
\item Χρειάζονται άπειρες ακολουθίες γεγονότων.
\item Κάθε πεπερασμένη ακολουθία δίνει ελαφρώς διαφορετικό αποτέλεσμα.
\end{itemize}
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Θεωρία πιθανοτήτων}
\begin{itemize}
\item Ανεξαρτήτως του τρόπου ερμηνείας της πιθανότητας τα μαθηματικά της μπορούν να οριστούν αξιωματικά (αξιωματική θεωρία πιθανοτήτων).
\item Έστω $\Omega$ το σύνολο των δυνατών εκβάσεων ενός πειράματος τύχης ({\it δειγματικός χώρος}).
\item Κάθε μονοσύνολο του $\Omega$ είναι ένα {\it απλό ενδεχόμενο}.
\item Κάθε ένωση απλών ενδεχομένων είναι ένα {\it σύνθετο ενδεχόμενο}.
\item Μέτρο πιθανότητας ή πιθανότητα $\pr$ είναι μια απεικόνιση $2^\Omega\rightarrow[0,1]$, τέτοια ώστε
\begin{enumerate}
\item $\pr(\Omega)=1$
\item Αν τα $a_1,\ldots,a_n$ είναι ξένα μεταξύ τους ενδεχόμενα, τότε\\
 $\pr(a_1\cup\cdots\cup a_n)=\pr(a_1)+\cdots+\pr(a_n)$
\end{enumerate}
\item Αν $\pr(a)=1$ το $a$ συμβαίνει πάντα (βέβαιο ενδεχόμενο), ενώ αν $\pr(a)=0$ το $a$ δεν συμβαίνει ποτέ (αδύνατο ενδεχόμενο).
\item H {\it κατανομή πιθανότητας} δίνει την πιθανότητα κάθε απλού ενδεχομένου.
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Βασικές ιδιότητες των πιθανοτήτων}
\begin{itemize}
\item $\pr(\bar{a})=1-\pr(a)$, όπου $\bar{a}$ είναι το συμπληρωματικό ενδεχόμενο του $a$.
\item $\pr(\emptyset)=0$.
\item $\pr(a\cup b)=\pr(a) + \pr(b) - \pr(a\cap b)$.
\item Αν τα $b_1,\ldots,b_n$ είναι ξένα μεταξύ τους ενδεχόμενα τέτοια ώστε $b_1\cup\cdots\cup b_n=\Omega$, τότε
\[\pr(a)=\pr(a\cap b_1) + \cdots + \pr(a\cap b_n)\]
\item Αν ο $\Omega$ είναι πεπερασμένος, πρακτικά βολεύει να αντιλαμβανόμαστε την πιθανότητα ως
\[\pr(a) = \displaystyle\frac{|\Omega(a)|}{|\Omega|}\]
(ο κλασικός ορισμός της πιθανότητας κατά \textlatin{Laplace}).
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Δεσμευμένη πιθανότητα}
\begin{itemize}
\item Η πιθανότητα ενός ενδεχομένου μπορεί να εξαρτάται από άλλα ενδεχόμενα.
\item Η {\it δεσμευμένη πιθανότητα} του $a$ δεδομένου του $b$ %, δηλαδή η πιθανότητα να συμβεί το $a$ δεδομένου ότι έχει συμβεί το $b$,
    είναι
\[\pr(a|b)\doteq\frac{\pr(a\cap b)}{\pr(b)}\]
\item Έπεται ότι
\[\pr(a\cap b)=\pr(a|b)\cdot\pr(b)\]
και γενικότερα ({\it κανόνας της αλυσίδας})
\[\pr(a_1\cap\cdots\cap a_n)=\pr(a_1|a_2\cap\cdots \cap a_n)\cdot\ldots\cdot\pr(a_{n-1}|a_n)\cdot\pr(a_n)\]
\item Τα $a$ και $b$ είναι {\it ανεξάρτητα} αν και μόνο αν \[\pr(a|b)=\pr(a)\quad \text{ ή }\quad \pr(b|a)=\pr(b)\quad \text{ ή }\quad \pr(a\cap b)=\pr(a)\cdot\pr(b)\]
\item Δεσμευμένες εκδοχές: π.χ. $\pr(\bar{a}|c)=1-\pr(a|c)$, $\pr(a|c)=\pr(a\cap b_1|c) + \cdots + \pr(a\cap b_n|c)$, $\pr(a|b\cap c)=\pr(a|c)$ (δεσμευμένη ανεξαρτησία).
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Κανόνας του \textlatin{Bayes}}
\begin{itemize}
\item Γνωρίζουμε ότι $\displaystyle\pr(a|b)\doteq\frac{\pr(a\cap b)}{\pr(b)}$ και $\displaystyle\pr(b|a)\doteq\frac{\pr(a\cap b)}{\pr(a)}$
\item Συνδυάζοντας τις δύο σχέσεις:
\[\pr(a|b)=\frac{\pr(a)\cdot\pr(b|a)}{\pr(b)}\]
Συσχετίζει την πιθανότητα του $a$ δεδομένου του $b$ με την πιθανότητα του $b$ δεδομένου του $a$.
\item Παράδειγμα: $\displaystyle\pr(\text{\color{red}ασθένεια}|\text{\color{red}σύμπτωμα})=\frac{\pr(\text{ασθένεια})\pr(\text{\color{red}σύμπτωμα}|\text{\color{red}ασθένεια})}{\pr(\text{σύμπτωμα})}$\\
\begin{itemize}
\item Ο υπολογισμός του $\pr(\text{ασθένεια}|\text{σύμπτωμα})$ ως στατιστικής συχνότητας είναι δύσκολος. Αφορά \textcolor{blue}{διαγνωστική σχέση}.
\item Μπορεί όμως να υπάρχουν στατιστικά στοιχεία για τα $\pr(\text{ασθένεια})$, $\pr(\text{σύμπτωμα})$ και $\pr(\text{σύμπτωμα}|\text{ασθένεια})$ (\textcolor{blue}{αιτιακή σχέση}) στον γενικό πληθυσμό.
\end{itemize}
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Παράδειγμα}
\begin{itemize}
\item $\displaystyle\pr(\text{μηνιγγίτιδα}|\text{αυχενική δυσκαμψία})=\frac{\pr(\text{μηνιγγίτιδα})\cdot\pr(\text{αυχενική δυσκαμψία}|\text{μηνιγγίτιδα})}{\pr(\text{αυχενική δυσκαμψία})}$
\item Έστω $\pr(\text{μηνιγγίτιδα})= 1/50.000$, $\pr(\text{αυχενική δυσκαμψία}) = 0,01$ και\\ $\pr(\text{αυχενική δυσκαμψία}|\text{μηνιγγίτιδα})=0,7$.
\item Αρα $\displaystyle\pr(\text{μηνιγγίτιδα}|\text{αυχενική δυσκαμψία})=\frac{1/50.000\cdot0,7}{0,01}=0,0014$.
\item Αν υπάρξει επιδημία μηνιγγίτιδας το $\pr(\text{μηνιγγίτιδα})$ θα αυξηθεί.
\begin{itemize}
\item Αν το $\pr(\text{μηνιγγίτιδα}|\text{αυχενική δυσκαμψία})$ έχει υπολογιστεί στατιστικά δεν υπάρχει τρόπος να επικαιροποιηθεί.
\item Αν έχει υπολογιστεί με τον τύπο, το $\pr(\text{μηνιγγίτιδα}|\text{αυχενική δυσκαμψία})$ θα αυξηθεί αναλογικά με το $\pr(\text{μηνιγγίτιδα})$.
\item Η αιτιακή σχέση $\pr(\text{αυχενική δυσκαμψία}|\text{μηνιγγίτιδα})$ δεν επηρεάζεται από την επιδημία.
\end{itemize}
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Συνδυασμός πληροφορίας}
\begin{itemize}
\item Έστω ότι έχουμε περισσότερα συμπτώματα, π.χ:
\begin{multline}\pr(\text{μηνιγγίτιδα}|\text{αυχενική δυσκαμψία}\wedge\text{εμετός})=\\\frac{\pr(\text{μηνιγγίτιδα})\,\cdot\,\pr(\text{αυχενική δυσκαμψία}\wedge\text{εμετός}|\text{μηνιγγίτιδα})}{\pr(\text{αυχενική δυσκαμψία}\wedge\text{εμετός})}\notag\end{multline}
\item Χρειαζόμαστε από κοινού κατανομές πιθανότητας.
\item Εναλλακτικά μπορούμε να κάνουμε απλουστευτικές υποθέσεις:
\begin{multline}\pr(\text{αυχενική δυσκαμψία}\wedge\text{εμετός}|\text{μηνιγγίτιδα})=\\\pr(\text{αυχενική δυσκαμψία}|\text{μηνιγγίτιδα})\pr(\text{εμετός}|\text{μηνιγγίτιδα})\notag\end{multline}
Η αυχενική δυσκαμψία και ο εμετός προκαλούνται από την μηνιγγίτιδα, αλλά δεν αλληλεπιδρούν.
\item {\it Δεσμευμένη ανεξαρτησία}
$\pr(x\wedge y|z)=\pr(x|z)\pr(y|z)$ ή\\[4pt]\phantom{x}\hskip6.34cm
$\pr(x|y\wedge z)=\pr(x|z)$ και $\pr(y|x\wedge z)=\pr(y|z)$
\item Η υπόθεση της δεσμευμένης ανεξαρτησίας καθιστά χρήσιμα τα πιθανοτικά συστήματα.
\end{itemize}
\end{small}
\vfill
\end{slide}


\begin{slide}
\begin{small}
{\large Απλοϊκός ταξινομητής \textlatin{Bayes}}
\begin{itemize}
\item Από το νόμο του \textlatin{Bayes} έχουμε:\vskip-30pt
\[\pr(c|x_1,\ldots,x_n)=\frac{\pr(x_1,\ldots,x_n|c)\cdot\pr(c)}{\pr(x_1,\ldots,x_n)}\]
\item Υπόθεση δεσμευμένης ανεξαρτησίας [απλοϊκότητα]: $\pr(x_1,\ldots,x_n|c)=\prod_i\pr(x_i|c)$
\item Αρα\vspace{-11pt}
\begin{align*}\pr(c|x_1,\ldots,x_n)=&\frac{\pr(c)}{\pr(x_1,\ldots,x_n)}\prod_i\pr(x_i|c)\\
&\propto \pr(c)\cdot\prod_i\pr(x_i|c)\label{}\end{align*}
\item Κανόνας ταξινόμησης (ποιο είναι το $c$?):
\[c=\arg\max_c \pr(c)\cdot\prod_i\pr(x_i|c)\]
\item Δεν χρειάζεται να γνωρίζουμε το $\pr(x_1,\ldots,x_n)$.
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Παράδειγμα}
\begin{itemize}
\item Θέλουμε να υπολογίσουμε την πιθανότερη ασθένεια δεδομένων κάποιων συμπτωμάτων.
\item Υπολογίζουμε με βάση τα δεδομένα τις εξής τιμές:
\begin{itemize}
\item $\pr(\text{\color{red}μηνιγγίτιδα})\cdot\pr(\text{αυχενική δυσκαμψία}|{\color{red}\text{μηνιγγίτιδα}})\cdot\pr(\text{εμετός}|\text{\color{red}μηνιγγίτιδα})$
\item $\pr(\text{\color{red}γρίπη})\cdot\pr(\text{αυχενική δυσκαμψία}|{\color{red}\text{γρίπη}})\cdot\pr(\text{εμετός}|\text{\color{red}γρίπη})$
\item $\pr(\text{\color{red}γαστρεντερίτιδα})\cdot\pr(\text{αυχενική δυσκαμψία}|{\color{red}\text{γαστρεντερίτιδα}})\cdot\pr(\text{εμετός}|\text{\color{red}γαστρεντερίτιδα})$
\item ...
\end{itemize}
\item Επιλέγουμε την ασθένεια που δίνει την μεγαλύτερη τιμή.
\item {\color{blue} Πιστεύουμε} (υποκειμενικά) ότι αυτή είναι η εκδηλωμένη ασθένεια στο βαθμό που προκύπτει από τα δεδομένα που διαθέτουμε.

\begin{small}
{\large Υπολογισμός υποκειμενικής πιθανότητας}
\begin{itemize}
\item Έστω ένα σύνολο προτάσεων $p_1,\ldots,p_n$ (στην προτασιακή λογική).\\[2pt]
$p_1$: Ο Μάριος είναι ψηλός, $p_2$: Η Ειρήνη είναι παντρεμένη, ...
\item Σε διαφορετικές καταστάσεις του κόσμου είναι αληθείς διαφορετικοί συνδυασμοί προτάσεων. Υπάρχουν $2^n$ δυνατοί κόσμοι.
\item Κάθε διαφορετική κατάσταση $i$ του κόσμου είναι μια ερμηνεία $\mathcal{I}_i$.
\begin{itemize}
\item Η ερμηνεία απεικονίζει κάθε πρόταση στην τιμή αληθές ή ψευδές.
\end{itemize}
\item Δεν γνωρίζουμε την πραγματική κατάσταση του κόσμου, πιστεύουμε όμως ότι καθένας από τους δυνατούς κόσμους $i$ είναι ο πραγματικός με βαθμό $J(\mathcal{I}_i)$.
\begin{enumerate}
\item $0\leq J(\mathcal{I}_i)\leq 1$.
\item  $\displaystyle\sum_i J(\mathcal{I}_i)= 1$.
\end{enumerate}
\item $J$ είναι η {\it από κοινού κατανομή πιθανότητας}. %Με βάση αυτή μπορούμε να υπολογίσουμε το βαθμό πίστης μας σε οποιαδήποτε πρόταση εμπλέκει τις προτασιακές μεταβλητές.
\end{itemize}
\end{small}

\begin{slide}
\begin{small}
{\large Υπολογισμός υποκειμενικής πιθανότητας}
\begin{itemize}
\item Ο βαθμός πίστης μας σε μια πρόταση $\alpha$:
\[\pr(\alpha)\doteq \sum_{\mathcal{I}\models \alpha} J(\mathcal{I})\]
δηλαδή το άθροισμα των βαθμών πίστης μας σε όλους τους κόσμους όπου η $\alpha$ είναι αληθής.
\item Και ασφαλώς ισχύει $\displaystyle\pr(\alpha|\beta)=\frac{\pr(\alpha\wedge\beta)}{\pr(\beta)}$.
\item Αρα
\begin{multline}
\pr(\text{Ψηλός(Μάριος)}|\text{Ανδρας(Μάριος)}\wedge \text{Κρητικός(Μάριος)})=\\
\frac{\sum_{\mathcal{I}\models \text{Ψηλός(Μάριος)}\wedge\text{Ανδρας(Μάριος)}\wedge \text{Κρητικός(Μάριος)}} J(\mathcal{I})}{\sum_{\mathcal{I}\models \text{Ανδρας(Μάριος)}\wedge \text{Κρητικός(Μάριος)}} J(\mathcal{I})}\notag
\end{multline}
\item Πώς μπορούμε όμως να γνωρίζουμε την από κοινού κατανομή πιθανότητας?
\begin{itemize}
\item Απαιτείται να γνωρίζουμε $2^n-1$ το πλήθος τιμές της $J(\mathcal{I})$.
\end{itemize}
\end{itemize}
\end{small}

\begin{slide}
\begin{small}
{\large Δίκτυα πίστης}
\begin{itemize}
\item Έστω $\mathcal{I}$ μια ερμηνεία των προτάσεων $p_1,\ldots,p_n$ την οποία συμβολίζουμε με $\langle P_1,\ldots,P_n\rangle$, όπου είτε $P_i = p_i$ είτε $P_i = \neg p_i$.
\item Αρα
\[J(\langle P_1,\ldots,P_n\rangle)=\pr(P_1\wedge P_2\wedge\cdots\wedge P_n)\]
\item Ακραία απλούστευση:
\begin{itemize}
\item Όλες οι ατομικές προτάσεις είναι ανεξάρτητες μεταξύ τους
\[J(\langle P_1,\ldots,P_n\rangle)=\pr(P_1)\cdot\pr(P_2)\cdots\pr(P_n)\]
\item Χρειαζόμαστε $n$ τιμές.
\item Συνήθως υπάρχουν εξαρτήσεις μεταξύ των προτάσεων που δεν μπορούμε να αγνοήσουμε.
\end{itemize}
\end{itemize}
\end{small}

\begin{slide}
\begin{small}
{\large Δίκτυα πίστης}
\begin{itemize}
\item Πιο ρεαλιστική απλούστευση:
\begin{itemize}
\item Κατασκευάζουμε ένα {\it δίκτυο πίστης}, δηλαδή ένα κατευθυντό άκυκλο γράφημα $(\mathcal{V},\mathcal{E})$ όπου

$\mathcal{V} = \{p_1,\ldots, p_n\}$ και

$\mathcal{E} = \{(p_i,p_j)\mid\text{αν θεωρούμε ότι η αλήθεια της } p_i \text{ επηρεάζει την αλήθεια της } p_j\}$

\item H $p_i$ είναι {\it γονέας} της $p_j$.
\end{itemize}
\item Έστω ότι διατάσσουμε τις προτάσεις έτσι ώστε αν η $p_i$ είναι γονέας της $p_j$, τότε $i<j$.
\item Με βάση τον κανόνα της αλυσίδας
\[J(\langle P_1,\ldots,P_n\rangle)=\pr(P_1)\cdot\pr(P_2|P_1)\cdot\pr(P_3|P_1\wedge P_2)\cdots\pr(P_n|P_1\wedge\cdots\wedge P_{n-1})\]
\item Πάλι χρειάζεται να γνωρίζουμε $2^n-1$ αριθμούς.
\end{itemize}

\begin{slide}
\begin{small}
{\large Δίκτυα πίστης}
\begin{itemize}
\item Μια ακόμη απλούστευση:
\begin{itemize}
\item \textcolor{red}{Κάθε προτασιακή μεταβλητή είναι δεσμευμένα ανεξάρτητη από τις λοιπές προηγούμενές της, δεδομένων των γονέων της}:
\[\pr(P_{j}|P_1\wedge\cdots\wedge P_{j-1})=\pr(P_j|\text{γονείς}(P_j))\]
όπου
\[\text{γονείς}(P_i)= \bigwedge_{(p_i,p_k)\in\mathcal{E}} P_k\]
είναι η σύζευξη των προτασιακών μεταβλητών $P_i$ που αντιστοιχούν στους γονείς της $p_i$ στο γράφημα με την υπόθεση ότι $\text{γονείς}(P_i)\subseteq\{P_{i-1},\ldots,P_1\}$.
\item Αρα, $J(\langle P_1,\ldots,P_n\rangle)=\pr(P_1|\text{γονείς}(P_1))\cdot\pr(P_2|\text{γονείς}(P_2))\cdot\pr(P_n|\text{γονείς}(P_n))$
\end{itemize}
\item Για κάθε $p$ με γονείς τις $s_1,\ldots s_k$ πρέπει να γνωρίζουμε τα
\[\pr(p|s_1\wedge\ldots\wedge s_k),\ \pr(p|\neg s_1\wedge\ldots\wedge s_k),\ \ldots, \ \pr(p|\neg s_1\wedge\ldots\wedge \neg s_k)\]
\begin{itemize}
\item Χρειαζόμαστε το πολύ $n\cdot 2^k$, όπου $k$ το μέγιστο πλήθος γονέων [$<<2^n$].
\end{itemize}
\end{itemize}

{\large Πίστη και ευλογοφάνεια}
\begin{itemize}
\item Έστω ότι βρίσκουμε και ρίχνουμε ένα νόμισμα.
\item Αν γνωρίζουμε ότι το νόμισμα είναι αμερόληπτο τότε ο βαθμός πίστης μας ότι θα έρθει κορώνα ή γράμματα είναι 0.5.
\item Αν δεν γνωρίζουμε αν το νόμισμα είναι αμερόληπτο ή όχι?
\begin{itemize}
\item Ο βαθμός πίστης μας ότι θα έρθει κορώνα ή γράμματα μπορεί να κυμαίνεται από 0 -- 1.
\item Δεν έχουμε λόγο να πιστέψουμε ότι θα έρθει κορώνα ή γράμματα.

$\bel(\text{κορώνα})=\bel(\text{γράμματα})=0$.
\item Δεν έχουμε λόγο να αμφιβάλουμε ότι θα έρθει κορώνα ή γράμματα.

$\pl(\text{κορώνα})=\pl(\text{γράμματα})=1$.
\end{itemize}
\item Το $\bel$ είναι η {\it πίστη}.
\item Το $\pl$ είναι η {\it ευλογοφάνεια}.
\item Θα μπορούσαμε να τα δούμε σαν άνω και κάτω όρια της πιθανότητας.
\end{itemize}

{\large Θεωρία \eng{Demspter-Shafer} ή θεωρία ενδείξεων}
\begin{itemize}
\item Έστω $\Omega$ το σύνολο των δυνατών καταστάσεων ενός συστήματος.
\item Το $2^\Omega$ είναι το σύνολο όλων των υποσυνόλων του $\Omega$ συμπεριλαμβανομένου του $\emptyset$ και του $\Omega$.
\begin{itemize}
\item Π.χ. αν $\Omega=\{a,b\}$ τότε $2^\Omega=\{\emptyset,\{a\},\{b\},\Omega\}$
\end{itemize}
\item {\it Μάζα πίστης} είναι μια απεικόνιση $m:2^\Omega\rightarrow [0,1]$, τέτοια ώστε:
\begin{itemize}
\item $m(\emptyset)$ = 0.
\item $\displaystyle\sum_{A\subseteq 2^\Omega} m(A)$ = 1.
\end{itemize}
\item Η μάζα του $A$ εκφράζει το ποσοστό των διαθέσιμων στοιχείων που υποστηρίζουν τον ισχυρισμό ότι η κατάσταση του συστήματος ανήκει στο $A$, {\it αλλά όχι} σε κάποιο υποσύνολο του $A$.

\end{itemize}

\begin{small}
{\large Ασαφή σύνολα}
\begin{itemize}
\item H θεωρία των ασαφών συνόλων εισήχθη από τον \textlatin{Lotfi Zadeh} το 1965.
\item Έστω ένα (κλασικό) σύνολο αντικειμένων $X$ ({\it καθολικό σύνολο} ή  {\it σύνολο αναφοράς}).
\item Ορίζουμε μια {\it συνάρτηση συμμετοχής}:
\[\mu_A:X\rightarrow[0,1]\]
\item To $(X,\mu_A)$ είναι το {\it ασαφές σύνολο} $A$. (Εναλλακτικός ορισμός: $A:X\rightarrow[0,1]$).
\item Όσο μεγαλύτερη η τιμή, τόσο περισσότερο έχει το $x$ την ιδιότητα που εκφράζει το $Α$.
\item Αν $\mu_A:X\rightarrow \{0,1\}$ τότε το $A$ είναι κλασικό σύνολο.
\begin{itemize}
    \item Αν $\mu_A(x) = 1$ τότε $x\in A$,\quad αν $\mu_A(x) = 0$ τότε $x\notin A$.
    \item H $\mu_A$ καλείται {\it χαρακτηριστική συνάρτηση}.
\end{itemize}
\item Αν το $X=\{x_1,\ldots,x_n\}$ είναι πεπερασμένο συχνά γράφουμε $A=\mu(x_1)/x_1+\ldots+\mu(x_n)/x_n$.
\end{itemize}
\end{small}

\begin{itemize}
\item {\it $\alpha$-τομή} του $Α$: $^\alpha A=\{x\mid A(x)\geq\alpha\}$.
\item  {\it Ισχυρή $\alpha$-τομή} του $Α$: $^{\alpha+} A=\{x\mid A(x)>\alpha\}$.
\item Για $\alpha_1<\alpha_2$: ${^{\alpha_1} A} \supseteq\ {^{\alpha_2} A}$ και ${^{\alpha_1+} A} \supseteq\ {^{\alpha_2+} A}$.
\item {\it Στήριξη} του $A$: $^{0+} A$.
\item {\it Πυρήνας} του $A$: $^{1} A$.
\item {\it Ύψος} του $A$: $h(A)=\max_{x\in X} A(x)$.
\begin{itemize}
\item To $A$ είναι {\it κανονικό} αν $h(A)=1$ και {\it υποκανονικό} όταν $h(A)<1$.
\end{itemize}
\item To $A$ επί του $\mathbb{R}$ είναι {\it κυρτό} αν και μόνο $Α(\lambda x_1+(1-\lambda)x_2)\geq \min[A(x_1),A(x_2)]$
για κάθε $x_1,x_2\mathbb{R}$ και $\lambda\in[0,1]$.
\item Σχέση υποσυνόλου:
\begin{itemize}
\item $A\subseteq Β$ αν και μόνο αν $\mu_A(x)\leq \mu_B(x)$ για κάθε $x\in X$.
\end{itemize}
\end{itemize}


\end{small}

\begin{slide}
\begin{small}
{\large Ιδιότητες}
\begin{itemize}
\item Εξ ορισμού ισχύουν:
\begin{itemize}
\item Αντιμεταθετικότητα: $A\cup B = B\cup A$,  $A\cap B = B\cap A$.
\item Προσεταιριστικότητα : $A\cup (B\cup C) = (A\cup B)\cup C$,  $A\cap (B\cap C) = (A\cap B)\cap C$.
\item Ουδέτερο στοιχείο: $A\cup \emptyset = A$,  $A\cap X = A$
\item Απορροφητικό στοιχείο: $A\cup X = X$,  $A\cap \emptyset = \emptyset$
\end{itemize}

\item Οι άλλες ιδιότητες των κλασικών συνόλων που εμπλέκουν περισσότερες από μία πράξεις εν γένει δεν ισχύουν για αυθαίρετους συνδυασμούς $t$-νορμών, $s$-νορμών και συμπληρωμάτων.

\item Μια $t$-νόρμα $i$ και $s$-νόρμα $u$ είναι {\it δυικές} ως προς το ασαφές συμπλήρωμα $c$ αν και μόνο αν:
\[c(i(a,b)) = u(c(a),c(b))\quad\text{και}\quad c(u(a,b)) = i(c(a),c(b))\]
δηλαδή αν ισχύουν οι κανόνες \textlatin{De Morgan}: $\overline{A\cap B} = \bar{A}\cup\bar{B}$, $\overline{A\cup B}. = \bar{A}\cap\bar{B}$
\end{itemize}

\end{small}

\item $\langle \min\{a,b\},\max\{a,b\}, 1-a\rangle$\quad\qquad ({\it συνήθεις τελεστές})
\item $\langle ab, a+b-ab, 1-a\rangle$
\item $\langle \max\{0,a+b-1\}, \min\{1,a+b\}, 1-a\rangle$
\item $\langle i_{min}(a,b), u_{max}(a,b), 1-a\rangle$
\end{itemize}

\item Εν γένει δεν ισχύουν:
\begin{enumerate}
\item Κανόνας της αντίφασης: $A\cap \bar{A} = \emptyset$
\item Κανόνας του αποκλεισμού του τρίτου: $A\cup \bar{A} = Ι$, όπου $I(x)=1$ για κάθε $x\in X$
\item Επιμεριστικότητα: $Α\cap (B\cup C)= (A\cap B) \cup (A\cap C)$
\end{enumerate}
\begin{itemize}
\item Θεώρημα: Αν μια δυϊκή τριάδα ικανοποιεί τις (1) και (2) δεν ικανοποιεί την (3).
\item Οι συνήθεις πράξεις ικανοποιούν την (3).
\end{itemize}

{\large Ασαφείς σχέσεις}
\begin{itemize}
\item Μια (κλασική) σχέση μεταξύ των συνόλων $X_1,\ldots, X_n$ είναι ένα υποσύνολο του $X_1\times\ldots\times X_n$.
\item Μια {\it ασαφής σχέση} επί των συνόλων $X_1,\ldots, X_n$ είναι ένα ασαφές σύνολο επί του $X_1\times\ldots\times X_n$.
    \[R(X_1,\ldots X_n): X_1\times\ldots \times X_n\rightarrow [0,1]\]
\item Ιδιαίτερα χρήσιμες είναι οι {\it δυαδικές} ασαφείς σχέσεις.
\begin{itemize}
\item Αναπαριστώνται με έναν {\it ασαφή σχεσιακό πίνακα}.
\end{itemize}
\item Παραδείγματα
\begin{itemize}
\item Πολύ μακριά: $P(U,V)$\vspace{-12pt}
\[
\begin{array}{cccc}
&&V\\
&&\text{Βοστώνη}&\text{Χονγκ Κονγκ}\\
U&\text{Σαν Φρανσίσκο}& 0.3 &0.9\\
&\text{Χονγκ Κονγκ}& 1 & 0\\
&\text{Τόκυο}& 0.95 & 0.1\\
\end{array}\]
\item Προσεγγιστικά ίσος: $P(\mathbb{R},\mathbb{R})$ με $P(x,y)=e^{-(x-y)^2}$
\end{itemize}
\end{itemize}
\end{small}

\begin{small}
{\large Σύνθεση ασαφών σχέσεων}
\begin{itemize}
\item Σύνθεση κλασικών σχέσεων : $P\circ Q = \{ \langle x,z\rangle \mid \langle x,y\rangle\in P \text{ και } \langle y,z\rangle\in Q\}$

\item Αν $P(X,Y)$ και $Q(Y,Z)$ είναι δύο δυαδικές ασαφείς σχέσεις επί των $X\times Y$ και $Y\times Z$, αντίστοιχα, τότε η {\it σύνθεσή} τους είναι η ασαφής σχέση
\[(P\circ Q)(x,z) = \sup_{y\in Y}\, i(P(x,y), Q(y,z))\]
επί του $X\times Z$, όπου $i$ είναι μια $t$-νόρμα.

\item Διάφορες συνθέσεις ανάλογα με την επιλογή της $i$
\begin{itemize}
\item Σύνθεση \textlatin{sup-min}: $(P\circ Q)(x,z) = \sup_{y\in Y}\, \min\{P(x,y), Q(y,z)\}$
\item Σύνθεση \textlatin{sup-product}: $(P\circ Q)(x,z) = \sup_{y\in Y}\, P(x,y)\cdot Q(y,z)$
\end{itemize}

\item Η σύνθεση δυϊκών σχέσεων μπορεί να υπολογιστεί ως το γινομένο σχεσιακών πινάκων.
\begin{itemize}
\item Πολλαπλασιασμός: $\min$ ή $\cdot$
\item Πρόσθεση: $\max$
\end{itemize}

\end{itemize}
\end{small}

\begin{itemize}
\item Σε μία λογική (προτασιακή/πρώτης τάξης) έχουμε
\begin{itemize}
\item {\it Σύνταξη}: Μια γλώσσα κατασκευής προτάσεων/εκφράσεων  (π.χ. $p$, $p(a)$, $\exists x. p(x)$, $p\rightarrow q$).
\item {\it Πεδίο τιμών αληθείας}: $\{\mathrm{True},\mathrm{False}\}$.
\item {\it Σημασιολογία}: Μηχανισμός απεικόνισης των πρότασεων/έκφρασεων στο πεδίο τιμών αληθείας (ερμηνεία).
\end{itemize}
\item Κάποιες προτάσεις αληθεύουν πάντα (σε κάθε ερμηνεία): {\it ταυτολογίες}.
\begin{itemize}

\item $(p \wedge (p\rightarrow q))\rightarrow q$
\end{itemize}

\item Ορίζουμε ένα σύνολο προτάσεων/εκφράσεων ({\it αξιώματα}) που θεωρούμε ότι αληθεύουν.
\begin{itemize}
\item $p$
\item $p\rightarrow q$

{\large Λογική \textlatin{Lukasiewitcz}}
\begin{itemize}
\item {\it Σημασιολογική συνέπεια}
\begin{itemize}
\item $\mathcal{K} \models \phi$, όπου $\mathcal{K}$ ένα σύνολο αξιωμάτων, αν $T(\phi)=1$ για κάθε ερμηνεία με $T(\mathcal{K})=1$.
\end{itemize}


\item {\it Συντακτική συνέπεια}
\begin{itemize}
\item $\mathcal{K} \vdash \phi$, όπου $\mathcal{K}$ ένα σύνολο αξιωμάτων, αν η $\phi$ προκύπτει από τα $\mathcal{A}$, $\mathcal{K}$, μέσω του \textlatin{modus ponens}.

\end{itemize}

\item Η λογική \textlatin{Lukasiewitcz} είναι (η μόνη) {\it αξιωματικοποιήσιμη} πολύτιμη/απειρότιμη λογική:
\[ \text{υπάρχει σύνολο αξιωμάτων $\mathcal{A}$: } \mathcal{K}\vdash \phi \text{ αν και μόνο αν } \mathcal{K}\models \phi \]

\item Αξιώματα $\mathcal{A}$:
\begin{align*}
A&\rightarrow (B\rightarrow A)\\
(A\rightarrow B)&\rightarrow ((B\rightarrow C)\rightarrow (A\rightarrow C))\\
((A\rightarrow B)\rightarrow B)&\rightarrow ((B\rightarrow A)\rightarrow A)\\
(\neg B\rightarrow \neg A)&\rightarrow (A\rightarrow B)
\end{align*}


\end{itemize}

\item Οι λεκτικές τιμές μπορεί να συνοδεύονται από χαρακτηρισμούς (πολύ, σχετικά, λίγο, κλπ).
\item Ένας χαρακτηρισμός που εφαρμόζεται σε μια λεκτική τιμή $Α$ (ασαφές σύνολο) είναι ένας {\it λεκτικός τροποποιητής $Η$}\vspace{-4pt}
\[\mbox{$Η$ $Α$}\]
\item Το αποτέλεσμα είναι ένα νέο ασαφές σύνολο.
\item Ο $H$ υλοποιείται από μια πράξη $h:[0,1]\rightarrow [0,1]$
\[Η A(x)=h(A(x))\]
\item H $h$ είναι αύξουσα, 1-1 και
\begin{itemize}
    \item {\it ισχυρή} αν $h(a)<a$,\ \ {\it ασθενής} αν $h(a)>a$
\end{itemize}
\item Παραδείγματα: {\it πολύ}: $h(a)=a^2$, {\it σχετικά}: $h(a)=\sqrt{a}$.
\item Συνήθης οικογένεια τροποποιητών: $h_\alpha(a)=a^\alpha$
\end{itemize}
\end{small}

\item Μια συνάρτηση $\mathcal{J}:[0,1]\times[0,1]\rightarrow [0,1]$ που ικανοποιεί κάποιες από τις
\begin{enumerate}

\item αν $a\leq b$ τότε $\mathcal{J}(x,a)\leq \mathcal{J}(x,b)$

\item $\mathcal{J}(a,b)=1$ αν και μόνο αν $a\leq b$
\item $\mathcal{J}(a,b)=\mathcal{J}(c(b),c(a))$ (αντίθεση)
\item $\mathcal{J}$ συνεχής συνάρτηση (συνέχεια)
\end{enumerate}

\end{itemize}

\begin{itemize}

\end{itemize}

\item {\bf Αποτίμηση κανόνα} για κάθε κανόνα $L^l$.
\begin{itemize}
\item Υπολογισμός της σχέσης $R^l(X_1,X_2,\ldots, X_n,Y)$ που αντιπροσωπεύει ο $L^l$.
\[R^l(x_1,x_2,\ldots,x_n,y)=\mathcal{J}(S^l(x_1,x_2,\ldots,x_n), B^l(y))\]
όπου $\mathcal{J}$ είναι ένας τελεστής ασαφούς συνεπαγωγής.
\end{itemize}
\end{enumerate}
\begin{itemize}
\item Στη συνέχεια οι κανόνες μπορούν να συνδυαστούν είτε πριν είτε μετά την πραγματοποίηση του συμπερασμού
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Συμπερασμός σε ασαφές σύστημα (ανά κανόνα)}

\begin{enumerate}
\setcounter{enumi}{2}
\item {\bf Πραγματοποίηση συμπερασμού} για  κάθε κανόνα $L^i$.
\begin{itemize}
\item Εφαρμογή του συνθετικού κανόνα συμπερασμού:
\[B_l'(y)=\sup_{\langle x_1,\ldots,x_n\rangle\in X_1\times\ldots X_n} i(A'(x_1,\ldots,x_n),R^l(x_1,x_2,\ldots, x_n,y))\]

\end{itemize}
\item {\bf Συνάθροιση κανόνων}.
\begin{itemize}
\item Συνδυασμός των αποτελεσμάτων των διαφόρων κανόνων.
\[B'(y)=u(B_1'(y),\ldots, B_k'(y))\]
όπου $u$ μια $s$-νόρμα, ή
\[B'(y)=i(B_1'(y),\ldots, B_k'(y))\]
όπου $i$ μια $t$-νόρμα.
\end{itemize}
\end{enumerate}
\end{small}
\vfill
\end{slide}

{\large Συμπερασμός σε ασαφές σύστημα (συνθετικά)}

\begin{enumerate}
\setcounter{enumi}{2}
\item {\bf Συνάθροιση κανόνων}.
\begin{itemize}
\item Συνδυασμός των σχέσεων που εκφράζουν οι διάφοροι κανόνες.
\[R(x_1,x_2,\ldots,x_n,y)=u(R^1(x_1,x_2,\ldots,x_n,y),\ldots, R^k(x_1,x_2,\ldots,x_n,y))\]
όπου $u$ μια $s$-νόρμα, ή
\[R(x_1,x_2,\ldots,x_n,y)=i(R^1(x_1,x_2,\ldots,x_n,y),\ldots, R^k(x_1,x_2,\ldots,x_n,y))\]
όπου $i$ μια $t$-νόρμα.
\end{itemize}
\item {\bf Πραγματοποίηση συμπερασμού} για  κάθε κανόνα $L^i$.
\begin{itemize}
\item Εφαρμογή του συνθετικού κανόνα συμπερασμού:
\[B'(y)=\sup_{\langle x_1,\ldots,x_n\rangle\in X_1\times\ldots X_n} i(A'(x_1,\ldots,x_n),R(x_1,x_2,\ldots, x_n,y))\]
\end{itemize}
\end{enumerate}

\begin{small}
{\large Ασαφή συστήματα: Παραδείγματα}

\begin{itemize}
\item {\bf Μηχανή συμπερασμού γινομένου}.
\begin{itemize}
\item Συλλογιστική ανά κανόνα με ένωση κανόνων, συνεπαγωγή \textlatin{Mamdani}, τομή αλγεβρικού γινομένου:
\[B'(y)=\max_{i=1\ldots k}\{
         \sup_{x_1,x_2,\ldots,x_n\in X_1\times X_2\times\ldots\times X_n}\{
           A'(x_1,\ldots,x_n)\prod_{i=1}^n A^l_i(x_i) B^l(y)\}\}\]

\item {\bf Μηχανή συμπερασμού ελαχίστου}.
\end{itemize}
\begin{itemize}
\item Συλλογιστική ανά κανόνα με ένωση κανόνων, συνεπαγωγή \textlatin{Mamdani}, τομή $\min$:
\[B'(y)=\max_{i=1\ldots k}\{
         \sup_{x_1,x_2,\ldots,x_n\in X_1\times X_2\times\ldots\times X_n}
           \min\{A'(x_1,\ldots,x_n),A^l_1(x_i),\ldots,A^l_n(x_n), B^l(y)\}\}\]

\end{itemize}
\end{itemize}
\end{small}

\begin{small}
{\large Ασαφή συστήματα: Παραδείγματα}

\begin{itemize}
\item {\bf Μηχανή συμπερασμού \textlatin{Lukasiewicz}}.
\begin{itemize}
\item Συλλογιστική ανά κανόνα με τομή κανόνων, συνεπαγωγή \textlatin{Lukasiewicz}, τομή $\min$:
\[B'(y)=\min_{i=1\ldots k}\{
         \sup_{x_1,x_2,\ldots,x_n\in X_1\times X_2\times\ldots\times X_n}
           \min\{ A'(x_1,\ldots,x_n),1-\min_{i=1}^n\{ A^l_i(x_i)\}+ B^l(y)\}\}\]
\end{itemize}
\item {\bf Μηχανή συμπερασμού \textlatin{Dienes-Rescher}}.
\begin{itemize}
\item Συλλογιστική ανά κανόνα με τομή κανόνων, συνεπαγωγή \textlatin{Dienes-Rescher}, τομή $\min$:
\[B'(y)=\min_{i=1\ldots k}\{
         \sup_{x_1,x_2,\ldots,x_n\in X_1\times X_2\times\ldots\times X_n}
           \min\{ A'(x_1,\ldots,x_n),\max\{1-\min_{i=1}^n\{ A^l_i(x_i)\}\}, B^l(y)\}\}\]
\end{itemize}
\end{itemize}
\end{small}

\begin{slide}
\begin{small}
{\large Ασαφοποίηση}
\begin{itemize}
\item Συνήθως οι είσοδοι ενός ασαφούς συστήματος είναι (μη ασαφείς) αριθμητικές τιμές.
\item Πριν από το πρώτο βήμα εκτελείται {\bf ασαφοποίηση εισόδων}: Μετατροπή της τιμής εισόδου $\hat{x}$ για τη μεταβλητή $\mathcal{X}_i$ σε ένα ασαφές σύνολο $A'$.
\begin{itemize}
\item {\it Σημειακή ασαφοποίηση}
\[A'_i(x)=\left\{\begin{array}{ll}1&\mbox{αν $x=\hat{x}$}\\0&\mbox{διαφορετικά}\end{array}\right.\]
\item {\it Γκαουσιανή ασαφοποίηση}
\[A'_i(x)=e^{-(\frac{x-\hat{x}}{a})},\quad a>0\]
\item {\it Τριγωνική ασαφοποίηση}
\[A'_i(x)=\left\{\begin{array}{ll}1-\frac{|x-\hat{x}|}{b}&\mbox{αν $|x-x_i|\leq b$}\\0&\mbox{διαφορετικά}\end{array}\right.,\quad  b>0\]
\end{itemize}
\end{itemize}
\end{small}
\vfill
\end{slide}

\begin{slide}
\begin{small}
{\large Αποασαφοποίηση}
\begin{itemize}
\item Η έξοδος ενός ασαφούς συστήματος είναι ένα ασαφές σύνολο.
\item Μετά το τελευταίο βήμα εκτελείται {\bf αποασαφοποίηση εισόδων}: Μετατροπή του ασαφούς συνόλου εξόδου $B'$ σε μια κλασική τιμή $\hat{y}$.
\begin{itemize}
\item {\bf Aποασαφοποίηση κέντρου μάζας}:\vspace{-6pt}
\[\hat{y}=\frac{\int_Y B'(y)y\,dy}{\int_Y B'(y)\,dy}\]
%\item {\bf Aποασαφοποίηση κέντρου μέσων}: (για $M$
\item {\bf Aποασαφοποίηση μέσου μεγίστων}:
\[\hat{y}=\frac{\int_{Y'}y\,dy}{\int_{Y'}\,dy},\quad Y'=\{y\mid B'(y)=\hat{\mu}\}, \hat{\mu}=\sup B'(y)\].
\end{itemize}
